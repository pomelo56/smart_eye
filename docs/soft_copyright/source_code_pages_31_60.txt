 10 |     /**
 11 |      * Returns the current camera permission status as a string compatible
 12 |      * with the Dart-side [PermissionStatus] enum.
 13 |      *
 14 |      * Values: "granted", "denied", "permanently_denied".
 15 |      *
 16 |      * The "permanently denied" case requires two conditions to both hold:
 17 |      * 1. The runtime check returns PERMISSION_DENIED.
 18 |      * 2. We have previously asked the user for the permission at least
 19 |      *    once (tracked in SharedPreferences), AND
 20 |      *    `shouldShowRequestPermissionRationale` returns false.
 21 |      *
 22 |      * On a fresh install, before we have ever asked the user,
 23 |      * `shouldShowRequestPermissionRationale` also returns false. Without
 24 |      * the persistent "has been requested" flag we would incorrectly
 25 |      * classify every first launch as permanently denied and jump straight
 26 |      * to the system settings page without ever showing the system
 27 |      * permission dialog.
 28 |      */
 29 |     private fun currentCameraPermissionStatus(): String {
 30 |         val granted = ContextCompat.checkSelfPermission(
 31 |             this, Manifest.permission.CAMERA
 32 |         ) == PackageManager.PERMISSION_GRANTED
 33 |         if (granted) return "granted"
 34 | 
 35 |         val hasRequestedBefore = prefs.getBoolean(cameraRequestedKey, false)
 36 |         if (!hasRequestedBefore) {
 37 |             // Never asked the user yet — the system dialog can still be shown.
 38 |             return "denied"
 39 |         }
 40 | 
 41 |         val rationale = ActivityCompat.shouldShowRequestPermissionRationale(
 42 |             this, Manifest.permission.CAMERA
 43 |         )
 44 |         // If rationale is false AND we have asked before, the user selected
 45 |         // "Don't ask again". The system dialog will not appear; the only
 46 |         // recovery path is the settings page.
 47 |         return if (rationale) "denied" else "permanently_denied"
 48 |     }
